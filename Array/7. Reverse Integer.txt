class Solution {
    public int reverse(int x) {
       int result = 0;
       int max = Integer.MAX_VALUE;
       int min = Integer.MIN_VALUE;
       while(x != 0){
        int digit = x%10;// take the last digit from x
        x/= 10; //remove last digit from x

        if(result > max /10){
            return 0;
        }else if(result < min / 10){
            return 0;
        }
        result = result * 10 + digit; 
       }
       return result;
    }
}
